Quick Examples

Basic Parsing

local json: SIMPLE_JSON value: detachable SIMPLE_JSON_VALUE do create json value := json.parse ('{"name": "Alice", "age": 30}') if attached value and then value.is_object then print (value.as_object.string_item ("name")) -- Alice print (value.as_object.integer_item ("age")) -- 30 end end

Building JSON with Fluent API

local json: SIMPLE_JSON obj: SIMPLE_JSON_OBJECT do create json obj := json.new_object .put_string ("Bob", "name") .put_integer (25, "age") .put_boolean (True, "active") print (obj.to_json_string) -- {"name":"Bob","age":25,"active":true} end

JSON Pointer Navigation

local pointer: SIMPLE_JSON_POINTER value: detachable SIMPLE_JSON_VALUE do create pointer.make ("/users/0/name") value := pointer.evaluate (json_doc) if attached value then print (value.to_json_string) end end

JSON Patch Operations

local patch: SIMPLE_JSON_PATCH result: SIMPLE_JSON_PATCH_RESULT do create patch patch.add_add_operation ("/email", json.string_value ("alice@example.com")) patch.add_replace_operation ("/age", json.integer_value (31)) result := patch.apply (document) if result.is_success then print (result.document.to_json_string) end end

Schema Validation

local schema: SIMPLE_JSON_SCHEMA result: SIMPLE_JSON_SCHEMA_VALIDATION_RESULT do create schema.make_from_string ('{ "type": "object", "required": ["name", "age"], "properties": { "name": {"type": "string"}, "age": {"type": "integer", "minimum": 0} } }') result := schema.validate (document) if result.is_valid then print ("Document is valid") else across result.errors as ic loop print (ic.message) end end end
