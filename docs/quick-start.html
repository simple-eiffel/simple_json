Quick Start Guide

Get up and running with SIMPLE_JSON in just 5 minutes!

Installation

Add SIMPLE_JSON to your project's ECF file:

<library name="simple_json" location="$SIMPLE_JSON/simple_json.ecf"/>

Basic Usage

1. Extract a Single Value

local
    json: JSON
    name: detachable STRING
do
    create json
    name := json.string ("{\"name\": \"Alice\"}", "name")
    
    if attached name as n then
        print ("Name: " + n)
    end
end

2. Access Nested Values

local
    json: JSON
    city: detachable STRING
do
    create json
    city := json.path_string (json_text, "user.address.city")
    
    if attached city as c then
        print ("City: " + c)
    end
end

3. Build JSON

local
    json: JSON
    json_string: STRING
do
    create json
    json_string := json.build
        .put_string ("name", "Bob")
        .put_integer ("age", 25)
        .put_boolean ("active", True)
        .to_string
    
    print (json_string)
    -- Output: {"name":"Bob","age":25,"active":true}
end

4. Validate JSON

local
    json: JSON
    user_input: STRING
do
    create json
    user_input := get_user_input
    
    if json.is_valid (user_input) then
        -- Safe to parse
        obj := json.parse (user_input)
    else
        print ("Invalid JSON!")
    end
end

Common Patterns

Task Method
Get string value json.string(text, "key")
Get integer value json.integer(text, "key")
Navigate nested json.path_string(text, "a.b.c")
Build JSON json.build.put_string(...).to_string
Validate json.is_valid(text)
